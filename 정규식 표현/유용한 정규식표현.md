### 유요한 정규식 표현
- 영어만
~~~ js
//영어만
/^[a-zA-Z]*$/; 

//숫자만
/^[0-9]*$/

// 영어+숫자 
/^[a-zA-Z0-9]*$/

//특수문자 포함 여부
/[~!@#$%^&*()_+|<>?:{}]/

//email
/^[0-9a-zA-Z]([-_\.]?[0-9a-zA-Z])*@[0-9a-zA-Z]([-_\.]?[0-9a-zA-Z])*\.[a-zA-Z]{2,3}$/i

//한글만
/[a-z0-9]|[ \[\]{}()<>?|`~!@#$%^&*-_+=,.;:\"'\\]/g

//영어만
/^[a-zA-Z]*$/

//전화번호 형식
/^\d{2,3}-\d{3,4}-\d{4}$/

// 영어 + 숫자
/^[a-zA-Z0-9]*$/

//주민등록번호
/\d{6} \- [1-4]\d{6}/

//영문+숫자+특수문자
/^(?=.*[A-Za-z])(?=.*\d)(?=.*[$@$!%*#?&])[A-Za-z\d$@$!%*#?&]{8,}$/

//URL
/^(https?:\/\/)?([\da-z\.-]+)\.([a-z\.]{2,6})([\/\w_\.-]*)*\/?$/

//IP Address
/^(?:(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)\.){3}(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)$/

//HTML Tag
/^<([a-z]+)([^<]+)*(?:>(.*)<\/\1>|\s+\/>)$/
~~~
